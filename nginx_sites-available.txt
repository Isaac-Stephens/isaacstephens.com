server {
    listen 80;
    server_name isaacstephens.com www.isaacstephens.com;

    # Serve static files directly
    location /static/ {
        alias /home/isaac/projects/isaacstephens.com/website/static/;
        expires 30d;
        access_log off;
    }

    # Proxy all other requests to Flask (Gunicorn)
    location / {
        proxy_pass http://127.0.0.1:5000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # Optional: improve error visibility
    error_log /var/log/nginx/isaacstephens_error.log;
    access_log /var/log/nginx/isaacstephens_access.log;
}
